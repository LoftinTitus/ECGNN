4/// Load CSV data from a file
5/// Returns a vector of vectors where each inner vector represents a row of float values
6pub fn load_csv(file_path: &str) -> Result<Vec<Vec<f64>>, Box<dyn std::error::Error>> {
7    let file = File::open(file_path)?;
8    let reader = BufReader::new(file);
9    let mut data = Vec::new();
10    
11    for (index, line) in reader.lines().enumerate() {
12        let line = line?;
13        
14        // Skip header (first line)
15        if index == 0 {
16            continue;
17        }
18        
19        if !line.trim().is_empty() {
20            let values: Result<Vec<f64>, _> = line
21                .split(',')
22                .map(|s| s.trim().parse::<f64>())
23                .collect();
24            
25            match values {
26                Ok(float_values) => data.push(float_values),
27                Err(e) => eprintln!("Error parsing line {}: {}", index + 1, e),
28            }
29        }
30    }
31    
32    Ok(data)
33}

35/// Load all CSV files from a folder
36/// Returns a vector containing all data from all CSV files in the folder
37pub fn load_all_data(folder_path: &str) -> Result<Vec<Vec<f64>>, Box<dyn std::error::Error>> {
38    let mut all_data = Vec::new();
39    
40    for entry in fs::read_dir(folder_path)? {
41        let entry = entry?;
42        let path = entry.path();
43        
44        if path.is_file() {
45            if let Some(extension) = path.extension() {
46                if extension == "csv" {
47                    if let Some(path_str) = path.to_str() {
48                        match load_csv(path_str) {
49                            Ok(mut file_data) => all_data.append(&mut file_data),
50                            Err(e) => eprintln!("Error loading file {:?}: {}", path, e),
51                        }
52                    }
53                }
54            }
55        }
56    }
57    
58    Ok(all_data)
59}

61/// Scale data by normalizing columns 2 and 3 (indices 1 and 2)
62/// Returns a new vector with scaled data
63pub fn data_scaling(data: &[Vec<f64>]) -> Vec<Vec<f64>> {
64    if data.is_empty() {
65        return Vec::new();
66    }
67    
68    // Extract columns 2 and 3 (indices 1 and 2)
69    let col2: Vec<f64> = data.iter().map(|row| row[1]).collect();
70    let col3: Vec<f64> = data.iter().map(|row| row[2]).collect();
71    
72    // Calculate means
73    let mean2 = col2.iter().sum::<f64>() / col2.len() as f64;
74    let mean3 = col3.iter().sum::<f64>() / col3.len() as f64;
75    
76    // Calculate variances
77    let variance2 = col2.iter().map(|x| (x - mean2).powi(2)).sum::<f64>() / col2.len() as f64;
78    let variance3 = col3.iter().map(|x| (x - mean3).powi(2)).sum::<f64>() / col3.len() as f64;
79    
80    // Create scaled dataset
81    let mut scaled_dataset = Vec::new();
82    
83    for row in data {
84        let scaled_row2 = if variance2 != 0.0 {
85            (row[1] - mean2) / variance2.sqrt()
86        } else {
87            0.0
88        };
89        
90        let scaled_row3 = if variance3 != 0.0 {
91            (row[2] - mean3) / variance3.sqrt()
92        } else {
93            0.0
94        };
95        
96        scaled_dataset.push(vec![row[0], scaled_row2, scaled_row3]);
97    }
98    
99    scaled_dataset
100}

102/// Segment data into chunks of specified length
103/// Returns a vector of segments, each containing segment_length rows
104pub fn data_segmentation(scaled_dataset: &[Vec<f64>], segment_length: usize) -> Vec<Vec<Vec<f64>>> {
105    let mut segments = Vec::new();
106    
107    for i in (0..scaled_dataset.len()).step_by(segment_length) {
108        let end_index = std::cmp::min(i + segment_length, scaled_dataset.len());
109        let segment = scaled_dataset[i..end_index].to_vec();
110        
111        // Only add segments that are exactly the specified length
112        if segment.len() == segment_length {
113            segments.push(segment);
114        }
115    }
116    
117    segments
118}
